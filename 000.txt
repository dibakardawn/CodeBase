<script>
/*--------------------------------------------------------------------------------------------*/
var percent = 60;
var rightStartFraction = 0;
var rightEndFraction = 0;
var leftStartFraction = 0;
var leftEndFraction = 0;
if(percent <= 50){
	rightStartFraction = 1.5;	
	rightEndFraction = rightStartFraction + (percent / 50);
	leftStartFraction = 0;
	leftEndFraction = 0;
}else if(percent > 50 && percent <= 100){
	rightStartFraction = 1.5;
	rightEndFraction = 2.5;
	leftStartFraction = 0;
	leftEndFraction = ((percent - 50) / 50) * 1.5;
	if(percent > 50 && percent < 56){
		leftEndFraction = leftEndFraction + 0.45;
	}else if(percent > 55 && percent < 61){
		leftEndFraction = leftEndFraction + 0.38;
	}else if(percent > 60 && percent < 67){
		leftEndFraction = leftEndFraction + 0.27;
	}
}
console.log("rightStartFraction : " + rightStartFraction + "; rightEndFraction : " + rightEndFraction + "; leftStartFraction : " + leftStartFraction + "; leftEndFraction : " + leftEndFraction);

/*var rightStartFraction = 1.5;
var rightEndFraction = 2.5;
var leftStartFraction = 0;
var leftEndFraction = 0.79;*/

//.3+
/*--------------------------------------------------------------------------------------------*/

var can = document.getElementById('canvas1');
var ctx = can.getContext('2d');

ctx.translate(50,50); // just to get it away from the edge


var greenPart = ctx.createLinearGradient(0,0,0,100);
greenPart.addColorStop(0, '#6b5019');
greenPart.addColorStop(1, '#8d6e35');

var whitePart = ctx.createLinearGradient(0,0,0,100);
whitePart.addColorStop(0, '#a89f69');
whitePart.addColorStop(1, '#8d6e35');


var width = 10;
ctx.lineWidth = width;

// First we make a clipping region for the left half
ctx.save();
ctx.beginPath();
ctx.rect(-width, -width, 50+width, 100 + width*2);
ctx.clip();

// Then we draw the left half
ctx.strokeStyle = greenPart;
ctx.beginPath();
ctx.lineCap = "round";
ctx.arc(50, 50, 50, Math.PI*leftStartFraction, Math.PI*leftEndFraction, false);
ctx.stroke();

ctx.restore(); // restore clipping region to default

// Then we make a clipping region for the right half
ctx.save();
ctx.beginPath();
ctx.rect(50, -width, 50+width, 100 + width*2);
ctx.clip();

// Then we draw the right half
ctx.strokeStyle = whitePart;
ctx.beginPath();
ctx.lineCap = "round";
ctx.arc(50, 50, 50, Math.PI*rightStartFraction, Math.PI*rightEndFraction, false);
ctx.stroke();

ctx.restore(); // restore clipping region to default

</script>
